<script>

const sections = [

  {

    ]

  },

  {

    desc:"Estimate what % of your recruiting/sourcing work week is spent on each activity. Sliders should add up to 100%.",

        {id:"s_search", lbl:"Searching for candidates (boolean, LinkedIn, etc.)"},

        {id:"s_inbound", lbl:"Reviewing & dispositioning inbound candidates"},

      ]}

    ]

  },

  {

    id:"quality", tag:"Sourcing quality", title:"Quality & confidence in sourcing",

    desc:"Help us understand how effective your current sourcing process feels.",

    questions:[

      {id:"pipeline_quality", type:"scale", text:"How would you rate the overall quality of candidates you're surfacing?", low:"Very low", high:"Very high"},

      {id:"time_to_slate", type:"radio", text:"On average, how long does it take you to build an initial candidate slate for a new req?", opts:["< 1 day","1–2 days","3–5 days","> 1 week"]}

    ]

  },

  {

    id:"pain", tag:"Friction & pain points", title:"Where do things slow down?",

    desc:"Identify the biggest drains on your productivity.",

    questions:[

      {id:"pain_checks", type:"checkbox", text:"Which of these are significant time drains for you? (select all that apply)", opts:[

        "Finding enough qualified candidates",

        "Building boolean/search strings",

        "Reviewing large volumes of unqualified profiles",

        "Personalizing outreach at scale",

        "Tracking outreach and follow-ups",

        "Getting hiring manager alignment on profiles",

        "Accessing talent intelligence / market data",

        "Deduplication across sources"

      ]},

      {id:"manual_hrs", type:"radio", text:"How many hours per week do you estimate you spend on tasks you'd consider repetitive or manual?", opts:["< 2 hrs","2–5 hrs","5–10 hrs","> 10 hrs"]},

      {id:"bottleneck", type:"open", text:"What is the single biggest bottleneck in your sourcing or recruiting workflow right now?"}

    ]

  },

  {

    id:"sat", tag:"Satisfaction & expectations", title:"Satisfaction & expectations",

    desc:"Tell us about your overall satisfaction and what you're hoping to get from the pilot.",

    questions:[

      {id:"tool_sat", type:"scale", text:"Overall, how satisfied are you with your current recruiting tool stack?", low:"Very dissatisfied", high:"Very satisfied"},

      {id:"findem_familiarity", type:"radio", text:"Before this pilot, how familiar were you with Findem?", opts:["Never heard of it","Heard of it but never used","Used it briefly","Regular user"]},

      {id:"expectations", type:"open", text:"What are you hoping Findem helps you accomplish over this pilot?"}

    ]

  },

  { id:"done" }

];


let cur = 0;

const ans = {};

const total_sections = sections.length - 1;


function pct() { return Math.round((cur / total_sections) * 100); }


function render() {

  const sec = sections[cur];

  document.getElementById("prog").style.width = pct() + "%";

  const el = document.getElementById("body");


  if (sec.id === "done") {

    el.innerHTML = `<div class="card"><div class="done-box">

      <div class="done-icon">✓</div>

      <div class="done-title">Thanks — you're all set!</div>

      <p class="done-desc" style="margin-top:0.5rem;">Your baseline responses have been recorded. We'll resurvey at month 3 and month 9 to track your progress with Findem.</p>

    </div></div>`;

    return;

  }


  let h = `<div class="card" style="margin-bottom:1rem;">

    <div class="section-tag">${sec.tag}</div>

    <div class="section-title">${sec.title}</div>

    <p class="section-desc" style="margin-top:6px;">${sec.desc}</p>

  </div>`;


  sec.questions.forEach(q => {

    h += `<div class="q-block"><div class="q-text">${q.text}</div>`;


    if (q.type === "radio") {

      h += `<div class="pill-group">`;

      q.opts.forEach(o => {

        const s = ans[q.id] === o ? " selected" : "";

        h += `<div class="pill${s}" onclick="pickRadio('${q.id}','${CSS.escape(o)}',this,event)">${o}</div>`;

      });

      h += `</div>`;

    }


    if (q.type === "checkbox") {

      h += `<div class="check-list">`;

      q.opts.forEach(o => {

        const c = (ans[q.id]||[]).includes(o) ? "checked" : "";

        h += `<label class="check-item"><input type="checkbox" ${c} onchange="toggleCheck('${q.id}','${o.replace(/'/g,"\\'")}',this)"> ${o}</label>`;

      });

      h += `</div>`;

    }


    if (q.type === "checkbox_other") {

      h += `<div class="check-list">`;

      q.opts.forEach(o => {

        const c = (ans[q.id]||[]).includes(o) ? "checked" : "";

        h += `<label class="check-item"><input type="checkbox" ${c} onchange="toggleCheck('${q.id}','${o}',this)"> ${o}</label>`;

      });

      const oc = (ans[q.id]||[]).includes("Other");

      h += `<label class="check-item"><input type="checkbox" ${oc?"checked":""} onchange="toggleOther('${q.id}',this)"> Other</label>`;

      h += `<input type="text" class="other-text" id="ot_${q.id}" placeholder="Please specify…" style="display:${oc?"block":"none"};" value="${ans[q.id+"_other"]||""}" oninput="ans['${q.id}_other']=this.value">`;

      h += `</div>`;

    }


    if (q.type === "scale") {

      const v = ans[q.id]||null;

      h += `<div class="scale-wrap"><div class="scale-btns">`;

      for(let i=1;i<=7;i++){

        const s = v===i?" selected":"";

        h += `<button class="scale-btn${s}" onclick="pickScale('${q.id}',${i},this)">${i}</button>`;

      }

      h += `</div><div class="scale-labels"><span>${q.low}</span><span>${q.high}</span></div></div>`;

    }


    if (q.type === "sliders") {

      h += `<div class="q-sub">Drag each slider — aim for a total of 100%</div>`;

      const ids = q.sliders.map(s=>s.id);

      q.sliders.forEach(s => {

        const v = ans[s.id]!==undefined ? ans[s.id] : 5;

        h += `<div class="slider-row">

          <span class="slider-lbl">${s.lbl}</span>

          <input type="range" min="0" max="100" step="1" value="${v}" oninput="slideUp('${s.id}',this.value,${JSON.stringify(ids)})">

          <span class="slider-lbl-val" id="sv_${s.id}">${v}%</span>

        </div>`;

      });

      const tot = ids.reduce((s,id)=>s+(ans[id]!==undefined?ans[id]:5),0);

      h += `<div class="total-row${tot>100?" over":""}" id="tot_sliders">Total: <b>${tot}%</b> ${tot===100?"✓":tot>100?"— please reduce to reach 100%":"— adjust to reach 100%"}</div>`;

    }


    if (q.type === "open") {

      h += `<textarea placeholder="Share your thoughts…" oninput="ans['${q.id}']=this.value">${ans[q.id]||""}</textarea>`;

    }


    h += `</div>`;

  });


  h += `<div class="nav">

    ${cur>0?`<button class="btn" onclick="go(-1)">← Back</button>`:`<span></span>`}

    <span class="nav-meta">Step ${cur+1} of ${total_sections}</span>

    <button class="btn btn-primary" onclick="go(1)">${cur===total_sections-1?"Submit →":"Continue →"}</button>

  </div>`;


  el.innerHTML = h;

  window.scrollTo({top:0,behavior:"smooth"});

}


function go(d) {

  cur = Math.max(0, Math.min(sections.length-1, cur+d));

  render();

}


function pickRadio(qid, val, el) {

  ans[qid] = el.textContent.trim();

  el.closest('.pill-group').querySelectorAll('.pill').forEach(p=>p.classList.remove('selected'));

  el.classList.add('selected');

}


function pickScale(qid, val, el) {

  ans[qid] = val;

  el.closest('.scale-btns').querySelectorAll('.scale-btn').forEach(b=>b.classList.remove('selected'));

  el.classList.add('selected');

}


function toggleCheck(qid, val, el) {

  if(!ans[qid]) ans[qid]=[];

  if(el.checked){ans[qid].push(val);}

  else{ans[qid]=ans[qid].filter(v=>v!==val);}

}


function toggleOther(qid, el) {

  const inp = document.getElementById('ot_'+qid);

  if(!ans[qid]) ans[qid]=[];

  if(el.checked){ ans[qid].push("Other"); inp.style.display="block"; }

  else{ ans[qid]=ans[qid].filter(v=>v!=="Other"); ans[qid+"_other"]=""; inp.style.display="none"; inp.value=""; }

}


function slideUp(sid, val, ids) {

  ans[sid] = parseInt(val);

  document.getElementById('sv_'+sid).textContent = val+'%';

  const tot = ids.reduce((s,id)=>s+(ans[id]!==undefined?ans[id]:5),0);

  const el = document.getElementById('tot_sliders');

  el.className = 'total-row'+(tot>100?' over':'');

  el.innerHTML = `Total: <b>${tot}%</b> ${tot===100?"✓":tot>100?"— please reduce to reach 100%":"— adjust to reach 100%"}`;

}


render();

</script>
